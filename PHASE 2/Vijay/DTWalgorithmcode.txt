import numpy as np

def dtw(x, y):
    m = len(x)
    n = len(y)
    DTW = np.zeros((m+1, n+1))
    DTW[:, 0] = np.inf
    DTW[0, :] = np.inf
    DTW[0, 0] = 0

    for i in range(1, m+1):
        for j in range(1, n+1):
            cost = abs(x[i-1] - y[j-1])
            DTW[i, j] = cost + min(DTW[i-1, j], DTW[i, j-1], DTW[i-1, j-1])

    return DTW[m, n]
